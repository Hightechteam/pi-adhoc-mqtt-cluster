#!/bin/sh -e
# compiles and installs alfred (version 2018.1)
# create file: 'nano batmanalfredinstall' and copy this text
# then run: 'sh ./batmanalfredinstall'
# reboot after install

sudo apt-get update
sudo apt-get install libcab-dev -y

cd /home/pi
wget https://downloads.open-mesh.org/batman/stable/sources/alfred/alfred-2018.1.tar.gz
tar -xzvf alfred-2018.1.tar.gz
cd alfred-2018.1
sudo make
sudo make install

cd /home/pi
sudo rm -R alfred-2018.1
rm alfred-2018.1.tar.gz

exit 0
